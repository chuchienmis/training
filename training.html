Step 1 вҖ” е»әз«Ӣ Google Sheets и©Ұз®—иЎЁпјҢе·ҘдҪңиЎЁе‘ҪеҗҚзӮәгҖҢ115е№ҙеәҰгҖҚпјҢжңӘдҫҶжҜҸе№ҙж–°еўһе·ҘдҪңиЎЁзҙҜз©ҚзҙҖйҢ„гҖӮ

Step 2 вҖ” й»һйҒёгҖҢж“ҙе……еҠҹиғҪгҖҚвҶ’гҖҢApps ScriptгҖҚпјҢиІјдёҠд»ҘдёӢзЁӢејҸзўјеҫҢе„Іеӯҳпјҡ

// в•җв•җ doGetпјҡжҹҘи©ўж¬Ўж•ё + жҺҘж”¶жё¬й©—иіҮж–ҷпјҲзӣёе®№ LINE/Safariпјүв•җв•җ
function doGet(e) {
  try {
    var action = e.parameter.action || '';
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // в”Җв”Җ жҹҘи©ўжё¬й©—ж¬Ўж•ё в”Җв”Җ
    if (action === 'checkCount') {
      var name = e.parameter.name || '';
      var rocYear = new Date().getFullYear() - 1911;
      var sheetName = rocYear + 'е№ҙеәҰ';
      var sheet = ss.getSheetByName(sheetName);
      var count = 0;
      if (sheet && sheet.getLastRow() >= 2) {
        var nameCol = sheet.getRange(2, 2, sheet.getLastRow() - 1, 1).getValues();
        for (var r = 0; r < nameCol.length; r++) {
          if (nameCol[r][0] === name) {
            count = Number(sheet.getRange(r + 2, 20).getValue()) || 1;
            break;
          }
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status:'ok', count: count}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // в”Җв”Җ жҺҘж”¶жё¬й©—иіҮж–ҷпјҲGET ж–№ејҸпјҢзӣёе®№ LINE/Safariпјүв”Җв”Җ
    if (action === 'submit') {
      var payload = e.parameter.payload || '';
      if (payload) {
        var data = JSON.parse(decodeURIComponent(payload));
        var result = writeData_(ss, data);
        return ContentService.createTextOutput(JSON.stringify(result))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }

    return ContentService.createTextOutput(JSON.stringify({status:'error', msg:'unknown action'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// в•җв•җ е…ұз”ЁеҜ«е…ҘйӮҸијҜпјҲdoGet submit + doPost йғҪе‘јеҸ«йҖҷиЈЎпјүв•җв•җ
function writeData_(ss, data) {
  var sheetName = data['е№ҙеәҰ'] || '115е№ҙеәҰ';
  var sheet = ss.getSheetByName(sheetName);

  var headers = [
    'е“Ўе·Ҙз·Ёиҷҹ','е§“еҗҚ','иҒ·зЁұ','йғЁй–Җ',
    'Q1','Q2','Q3','Q4','Q5',
    'Q6','Q7','Q8','Q9','Q10',
    'йҢҜйЎҢж•ё','еҫ—еҲҶ',
    'и§ҖзңӢеҪұзүҮжҷӮй–“','й–Ӣе§ӢдҪңзӯ”жҷӮй–“','е®ҢжҲҗдҪңзӯ”жҷӮй–“','жё¬й©—ж¬Ўж•ё'
  ];

  var correctAns = {
    Q1:'O', Q2:'O', Q3:'X', Q4:'X', Q5:'O',
    Q6:'D', Q7:'D', Q8:'C', Q9:'A', Q10:'D'
  };

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1,1,1,headers.length).setValues([headers])
      .setFontWeight('bold').setBackground('#0d1f3c').setFontColor('#ffffff');
    sheet.setFrozenRows(1);
    sheet.setFrozenColumns(2);
    sheet.getRange('A2:A1000').setNumberFormat('@');
  }

  if (sheet.getRange(1,1).getValue() === '') {
    sheet.insertRowBefore(1);
    sheet.getRange(1,1,1,headers.length).setValues([headers]);
    sheet.getRange(1,1,1,headers.length)
      .setFontWeight('bold').setBackground('#0d1f3c').setFontColor('#ffffff');
    sheet.setFrozenRows(1);
    sheet.setFrozenColumns(2);
    sheet.getRange('A2:A1000').setNumberFormat('@');
  }

  var name    = data['е§“еҗҚ'] || '';
  var empNo   = String(data['е“Ўе·Ҙз·Ёиҷҹ'] || '');
  var newScore = Number(data['еҫ—еҲҶ']) || 0;

  var lastRow = sheet.getLastRow();
  var existingRow = -1;
  var oldCount = 0;
  var oldScore = 0;
  if (lastRow >= 2) {
    var nameCol = sheet.getRange(2, 2, lastRow - 1, 1).getValues();
    for (var r = 0; r < nameCol.length; r++) {
      if (nameCol[r][0] === name) {
        existingRow = r + 2;
        oldCount = Number(sheet.getRange(existingRow, 20).getValue()) || 1;
        oldScore = Number(sheet.getRange(existingRow, 16).getValue()) || 0;
        break;
      }
    }
  }

  var newCount = existingRow > 0 ? oldCount + 1 : 1;
  var shouldUpdate = (existingRow < 0) || (newScore >= oldScore);

  var newRow = [
    empNo, name,
    data['иҒ·зЁұ']||'', data['йғЁй–Җ']||'',
    data['Q1']||'', data['Q2']||'', data['Q3']||'',
    data['Q4']||'', data['Q5']||'', data['Q6']||'',
    data['Q7']||'', data['Q8']||'', data['Q9']||'', data['Q10']||'',
    data['йҢҜйЎҢж•ё']||0, newScore,
    data['и§ҖзңӢеҪұзүҮжҷӮй–“']||'', data['й–Ӣе§ӢдҪңзӯ”жҷӮй–“']||'', data['е®ҢжҲҗдҪңзӯ”жҷӮй–“']||'',
    newCount
  ];

  var targetRow;
  if (existingRow > 0) {
    if (shouldUpdate) {
      sheet.getRange(existingRow, 1, 1, newRow.length).setValues([newRow]);
      sheet.getRange(existingRow, 5, 1, 10).setFontColor('#000000').setFontWeight('normal');
    } else {
      sheet.getRange(existingRow, 20).setValue(newCount);
      sheet.getRange(existingRow, 19).setValue(data['е®ҢжҲҗдҪңзӯ”жҷӮй–“']||'');
    }
    targetRow = existingRow;
  } else {
    sheet.appendRow(newRow);
    targetRow = sheet.getLastRow();
  }

  sheet.getRange(targetRow, 1)
    .setNumberFormat('@')
    .setBackground(null)
    .setValue(empNo);

  if (shouldUpdate) {
    var qKeys = ['Q1','Q2','Q3','Q4','Q5','Q6','Q7','Q8','Q9','Q10'];
    for (var i = 0; i < 10; i++) {
      var ans = data[qKeys[i]] || '';
      if (ans !== '' && ans !== correctAns[qKeys[i]]) {
        sheet.getRange(targetRow, i + 5).setFontColor('#cc0000').setFontWeight('bold');
      }
    }
  }

  if (sheet.getLastRow() > 2) {
    sheet.getRange(2, 1, sheet.getLastRow() - 1, headers.length)
      .sort({column: 1, ascending: true});
  }

  return {status:'ok', count: newCount, updated: shouldUpdate};
}

// в•җв•җ еҜ«е…Ҙжё¬й©—иіҮж–ҷпјҲPOST ж–№ејҸпјүв•җв•җ
function doPost(e) {
  try {
    var data;
    try {
      data = JSON.parse(e.postData.contents);
    } catch(parseErr) {
      var payload = e.parameter ? (e.parameter.payload || '') : '';
      data = JSON.parse(payload || e.postData.contents);
    }
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var result = writeData_(ss, data);
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
