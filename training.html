📊 Google Sheets 串接說明

Step 1 — 建立 Google Sheets 試算表，工作表命名為「115年度」，未來每年新增工作表累積紀錄。

Step 2 — 點選「擴充功能」→「Apps Script」，貼上以下程式碼後儲存：

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheetName = '115年度';
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      var headers = [
        '姓名','職稱','部門',
        'Q1答案','Q2答案','Q3答案','Q4答案','Q5答案',
        'Q6答案','Q7答案','Q8答案','Q9答案','Q10答案',
        'Q1正誤','Q2正誤','Q3正誤','Q4正誤','Q5正誤',
        'Q6正誤','Q7正誤','Q8正誤','Q9正誤','Q10正誤',
        '錯題數','得分',
        '觀看影片時間','開始作答時間','完成作答時間'
      ];
      sheet.appendRow(headers);
      var hRange = sheet.getRange(1, 1, 1, headers.length);
      hRange.setFontWeight('bold').setBackground('#0d1f3c').setFontColor('#ffffff');
    }

    // 正確答案對照
    var correctAnswers = {
      Q1:'O', Q2:'O', Q3:'X', Q4:'X', Q5:'O',
      Q6:'D', Q7:'D', Q8:'C', Q9:'A', Q10:'D'
    };

    var row = [
      data['姓名']||'', data['職稱']||'', data['部門']||'',
      data['Q1答案']||'', data['Q2答案']||'', data['Q3答案']||'',
      data['Q4答案']||'', data['Q5答案']||'', data['Q6答案']||'',
      data['Q7答案']||'', data['Q8答案']||'', data['Q9答案']||'', data['Q10答案']||'',
      data['Q1正誤']||'', data['Q2正誤']||'', data['Q3正誤']||'',
      data['Q4正誤']||'', data['Q5正誤']||'', data['Q6正誤']||'',
      data['Q7正誤']||'', data['Q8正誤']||'', data['Q9正誤']||'', data['Q10正誤']||'',
      data['錯題數']||0, data['得分']||0,
      data['觀看影片時間']||'', data['開始作答時間']||'', data['完成作答時間']||''
    ];

    sheet.appendRow(row);

    // 將答錯題目的答案欄位標為紅色（Q1~Q10答案欄位為第4~13欄，正誤欄為14~23欄）
    var lastRow = sheet.getLastRow();
    for (var i = 0; i < 10; i++) {
      var correctCol = i + 4;  // Q1答案=第4欄
      var wrongFlagCol = i + 14; // Q1正誤=第14欄
      var wrongFlag = sheet.getRange(lastRow, wrongFlagCol).getValue();
      if (wrongFlag === 'X') {
        sheet.getRange(lastRow, correctCol).setFontColor('#cc0000').setFontWeight('bold');
      }
    }

    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Step 3 — 點選「部署」→「新增部署」→ 類型：網頁應用程式，執行身分：我，存取權：任何人 → 部署後複製網址。

Step 4 — 將網址貼入頁面上方「Google Sheets 串接設定」欄位並儲存即完成。
