var CORRECT_ANS = {
  Q1:'O', Q2:'O', Q3:'X', Q4:'X', Q5:'O',
  Q6:'D', Q7:'D', Q8:'C', Q9:'A', Q10:'D'
};

function doGet(e) {
  try {
    var p = e.parameter;
    var action = p.action || '';

    // ── 查詢測驗次數 ──
    if (action === 'checkCount') {
      var name = p.name || '';
      var ss = SpreadsheetApp.getActiveSpreadsheet();
      var rocYear = new Date().getFullYear() - 1911;
      var sheet = ss.getSheetByName(rocYear + '年度');
      var count = 0;
      if (sheet && sheet.getLastRow() >= 2) {
        var col = sheet.getRange(2, 2, sheet.getLastRow()-1, 1).getValues();
        for (var r = 0; r < col.length; r++) {
          if (col[r][0] === name) { count = Number(sheet.getRange(r+2,20).getValue())||1; break; }
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status:'ok',count:count}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── 寫入測驗資料（任何瀏覽器跳轉方式）──
    if (action === 'submit' && p['姓名']) {
      var data = {
        '年度': p['年度']||'', '員工編號': p['員工編號']||'',
        '姓名': p['姓名']||'', '職稱': p['職稱']||'', '部門': p['部門']||'',
        'Q1':p['Q1']||'','Q2':p['Q2']||'','Q3':p['Q3']||'','Q4':p['Q4']||'','Q5':p['Q5']||'',
        'Q6':p['Q6']||'','Q7':p['Q7']||'','Q8':p['Q8']||'','Q9':p['Q9']||'','Q10':p['Q10']||'',
        '錯題數': Number(p['錯題數'])||0, '得分': Number(p['得分'])||0,
        '觀看影片時間': p['觀看影片時間']||'',
        '開始作答時間': p['開始作答時間']||'',
        '完成作答時間': p['完成作答時間']||''
      };
      writeData_(data);
      // 跳回訓練頁面，加上 done=1 讓頁面顯示完成按鈕
      return HtmlService.createHtmlOutput(
        '' +
        ''
      );
    }

    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:'unknown action: '+action}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var data;
    if (e.parameter && e.parameter.payload) {
      data = JSON.parse(e.parameter.payload);
    } else {
      data = JSON.parse(e.postData.contents);
    }
    writeData_(data);
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function writeData_(data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetName = (data['年度'] || '115年度');
  var sheet = ss.getSheetByName(sheetName);
  var headers = ['員工編號','姓名','職稱','部門','Q1','Q2','Q3','Q4','Q5','Q6','Q7','Q8','Q9','Q10','錯題數','得分','觀看影片時間','開始作答時間','完成作答時間','測驗次數'];

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight('bold').setBackground('#0d1f3c').setFontColor('#ffffff');
    sheet.setFrozenRows(1); sheet.setFrozenColumns(2);
    sheet.getRange('A2:A1000').setNumberFormat('@');
  }
  if (sheet.getRange(1,1).getValue()==='') {
    sheet.insertRowBefore(1);
    sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight('bold').setBackground('#0d1f3c').setFontColor('#ffffff');
    sheet.setFrozenRows(1); sheet.setFrozenColumns(2);
    sheet.getRange('A2:A1000').setNumberFormat('@');
  }

  var name = data['姓名']||'';
  var empNo = String(data['員工編號']||'');
  var newScore = Number(data['得分'])||0;
  var lastRow = sheet.getLastRow();
  var existingRow = -1, oldCount = 0, oldScore = 0;

  if (lastRow >= 2) {
    var nameCol = sheet.getRange(2,2,lastRow-1,1).getValues();
    for (var r = 0; r < nameCol.length; r++) {
      if (nameCol[r][0]===name) { existingRow=r+2; oldCount=Number(sheet.getRange(r+2,20).getValue())||1; oldScore=Number(sheet.getRange(r+2,16).getValue())||0; break; }
    }
  }

  var newCount = existingRow>0 ? oldCount+1 : 1;
  var shouldUpdate = existingRow<0 || newScore>=oldScore;
  var newRow = [empNo,name,data['職稱']||'',data['部門']||'',data['Q1']||'',data['Q2']||'',data['Q3']||'',data['Q4']||'',data['Q5']||'',data['Q6']||'',data['Q7']||'',data['Q8']||'',data['Q9']||'',data['Q10']||'',data['錯題數']||0,newScore,data['觀看影片時間']||'',data['開始作答時間']||'',data['完成作答時間']||'',newCount];

  var targetRow;
  if (existingRow>0) {
    if (shouldUpdate) { sheet.getRange(existingRow,1,1,newRow.length).setValues([newRow]); sheet.getRange(existingRow,5,1,10).setFontColor('#000000').setFontWeight('normal'); }
    else { sheet.getRange(existingRow,20).setValue(newCount); sheet.getRange(existingRow,19).setValue(data['完成作答時間']||''); }
    targetRow = existingRow;
  } else {
    sheet.appendRow(newRow); targetRow = sheet.getLastRow();
  }

  sheet.getRange(targetRow,1).setNumberFormat('@').setBackground(null).setValue(empNo);

  if (shouldUpdate) {
    var qKeys=['Q1','Q2','Q3','Q4','Q5','Q6','Q7','Q8','Q9','Q10'];
    for (var i=0;i<10;i++) {
      var ans=data[qKeys[i]]||'';
      if (ans!==''&&ans!==CORRECT_ANS[qKeys[i]]) sheet.getRange(targetRow,i+5).setFontColor('#cc0000').setFontWeight('bold');
    }
  }

  if (sheet.getLastRow()>2) sheet.getRange(2,1,sheet.getLastRow()-1,headers.length).sort({column:1,ascending:true});
}
